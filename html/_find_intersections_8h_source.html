123 Point intersectionOf(Line l1, Line l2){
124 Point intersect;
125 if(checkIntersect(l1, l2) == 0){
126 intersect.x = -1;
127 intersect.y = -1;
128 }else{
129 double a1 = l1.lower.y - l1.upper.y;
130 double b1 = l1.upper.x - l1.lower.x;
131 double c1 = a1*(l1.upper.x) + b1*(l1.upper.y);
132
133 double a2 = l2.lower.y - l2.upper.y;
134 double b2 = l2.upper.x - l2.lower.x;
135 double c2 = a2*(l2.upper.x) + b2*(l2.upper.y);
136
137 double determinant = a1*b2 - a2*b1;
138 intersect.x = (b2*c1 - b1*c2)/determinant;
139 intersect.y = (a1*c2 - a2*c1)/determinant;
140 }
141 return intersect;
